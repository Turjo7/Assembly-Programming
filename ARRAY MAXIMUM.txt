.MODEL SMALL

.STACK 100H

.DATA    

    array DB 2,8,9,5,7

.CODE


MAIN PROC   
    
    MOV AX,@DATA
    MOV DS,AX 
    
    MOV CX,5
    MOV DI,0
      
    SUB AL,AL
    
    BIG: 
    CMP AL,array[DI]
    
    JA NEXT                      ; WORKS FOR JAE, JNLE
    
    MOV AL,array[DI]
    
    NEXT:
    INC DI
    LOOP BIG
    
    MOV AH,2
    ADD AL,30H
    MOV DL,AL
    INT 21H
    
    
    
    
    
    MAIN ENDP
END MAIN
